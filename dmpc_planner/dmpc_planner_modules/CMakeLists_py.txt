cmake_minimum_required(VERSION 3.8)
project(dmpc_planner_modules)

## Find catkin and declare dependencies
find_package(catkin REQUIRED COMPONENTS
  rospy
  std_msgs
  mpc_planner_types
  mpc_planner_util
  mpc_planner_solver
  ros_tools
)

## Declare catkin package
catkin_package(
  CATKIN_DEPENDS rospy std_msgs mpc_planner_types mpc_planner_util mpc_planner_solver ros_tools
)

## Install Python scripts
catkin_install_python(
  PROGRAMS
    scripts/mpc_planner_node.py
    scripts/mpc_planner_utils.py
  DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)

## Install Python modules (if any)
install(DIRECTORY src/${PROJECT_NAME}/
  DESTINATION ${CATKIN_PACKAGE_PYTHON_DESTINATION}
  FILES_MATCHING PATTERN "*.py"
)

## Install configuration files or launch files
install(DIRECTORY launch/
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/launch
  FILES_MATCHING PATTERN "*.launch"
)
